void set(char arr[N][N])
{
	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < N; j++)
		{
			arr[i][j] = '-';
		}
	}
}
void print(char arr[N][N])
{
	cout << "\n\n\n\n\n";
	for (int i = 0; i < N; i++)
	{
		for (int j = 0; j < N; j++)
		{
			cout << setw(5) << arr[i][j];
		}
		cout << "\n";
	}
	cout << "\n\n\n\n\n";
}
